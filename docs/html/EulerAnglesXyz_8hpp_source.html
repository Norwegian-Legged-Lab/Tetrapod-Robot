263  EulerAnglesXyz getUnique() const {
264  Base xyz(kindr::floatingPointModulo(x()+M_PI,2*M_PI)-M_PI,
265  kindr::floatingPointModulo(y()+M_PI,2*M_PI)-M_PI,
266  kindr::floatingPointModulo(z()+M_PI,2*M_PI)-M_PI); // wrap all angles into [-pi,pi)
267 
268  const double tol = 1e-3;
269 
270  // wrap angles into [-pi,pi),[-pi/2,pi/2),[-pi,pi)
271  if(xyz.y() < -M_PI/2 - tol)
272  {
273  if(xyz.x() < 0) {
274  xyz.x() = xyz.x() + M_PI;
275  } else {
276  xyz.x() = xyz.x() - M_PI;
277  }
278 
279  xyz.y() = -(xyz.y() + M_PI);
280 
281  if(xyz.z() < 0) {
282  xyz.z() = xyz.z() + M_PI;
283  } else {
284  xyz.z() = xyz.z() - M_PI;
285  }
286  }
287  else if(-M_PI/2 - tol <= xyz.y() && xyz.y() <= -M_PI/2 + tol)
288  {
289  xyz.x() -= xyz.z();
290  xyz.z() = 0;
291  }
292  else if(-M_PI/2 + tol < xyz.y() && xyz.y() < M_PI/2 - tol)
293  {
294  // ok
295  }
296  else if(M_PI/2 - tol <= xyz.y() && xyz.y() <= M_PI/2 + tol)
297  {
298  // todo: M_PI/2 should not be in range, other formula?
299  xyz.x() += xyz.z();
300  xyz.z() = 0;
301  }
302  else // M_PI/2 + tol < xyz.y()
303  {
304  if(xyz.x() < 0) {
305  xyz.x() = xyz.x() + M_PI;
306  } else {
307  xyz.x() = xyz.x() - M_PI;
308  }
309 
310  xyz.y() = -(xyz.y() - M_PI);
311 
312  if(xyz.z() < 0) {
313  xyz.z() = xyz.z() + M_PI;
314  } else {
315  xyz.z() = xyz.z() - M_PI;
316  }
317  }
318 
319  return EulerAnglesXyz(xyz);
320  }

330  typename Eigen::Matrix<PrimType_, 3, 3> getMappingFromDiffToLocalAngularVelocity() const {
331  using std::sin;
332  using std::cos;
333  Eigen::Matrix<PrimType_, 3, 3> mat = Eigen::Matrix<PrimType_, 3, 3>::Zero();
334  const PrimType_ x = this->x();
335  const PrimType_ y = this->y();
336  const PrimType_ z = this->z();
337 
338  const PrimType_ t2 = cos(y);
339  const PrimType_ t3 = sin(z);
340  const PrimType_ t4 = cos(z);
341  mat(0, 0) = t2*t4;
342  mat(0, 1) = t3;
343  mat(1, 0) = -t2*t3;
344  mat(1, 1) = t4;
345  mat(2, 0) = sin(y);
346  mat(2, 2) = 1.0;
347 
348  return mat;
349  }

351  typename Eigen::Matrix<PrimType_, 3, 3> getMappingFromLocalAngularVelocityToDiff() const {
352  using std::sin;
353  using std::cos;
354  Eigen::Matrix<PrimType_, 3, 3> mat = Eigen::Matrix<PrimType_, 3, 3>::Zero();
355  const PrimType_ x = this->x();
356  const PrimType_ y = this->y();
357  const PrimType_ z = this->z();
358 
359  const PrimType_ t2 = cos(y);
360  KINDR_ASSERT_TRUE(std::runtime_error, t2 != PrimType_(0.0), "Gimbal lock: cos(y) is zero!");
361  const PrimType_ t3 = 1.0/t2;
362  const PrimType_ t4 = sin(z);
363  const PrimType_ t5 = cos(z);
364  const PrimType_ t6 = sin(y);
365  mat(0,0) = t3*t5;
366  mat(0,1) = -t3*t4;
367  mat(1,0) = t4;
368  mat(1,1) = t5;
369  mat(2,0) = -t3*t5*t6;
370  mat(2,1) = t3*t4*t6;
371  mat(2,2) = 1.0;
372 
373  return mat;
374  }

376  typename Eigen::Matrix<PrimType_, 3, 3> getMappingFromGlobalAngularVelocityToDiff() const {
377  using std::sin;
378  using std::cos;
379  Eigen::Matrix<PrimType_, 3, 3> mat = Eigen::Matrix<PrimType_, 3, 3>::Zero();
380  const PrimType_ x = this->x();
381  const PrimType_ y = this->y();
382  const PrimType_ z = this->z();
383  const PrimType_ t2 = cos(y);
384  KINDR_ASSERT_TRUE(std::runtime_error, t2 != PrimType_(0.0), "Gimbal lock: cos(y) is zero!");
385  const PrimType_ t3 = 1.0/t2;
386  const PrimType_ t4 = sin(y);
387  const PrimType_ t5 = cos(x);
388  const PrimType_ t6 = sin(x);
389  mat(0,0) = 1.0;
390  mat(0,1) = t3*t4*t6;
391  mat(0,2) = -t3*t4*t5;
392  mat(1,1) = t5;
393  mat(1,2) = t6;
394  mat(2,1) = -t3*t6;
395  mat(2,2) = t3*t5;
396  return mat;
397  }
