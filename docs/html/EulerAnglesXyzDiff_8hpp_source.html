239  Eigen::Matrix<PrimType_, 3, 3> getMappingFromLocalAngularVelocityToSecondDiff(const EulerAnglesXyz<PrimType_>& rotation) const {
240  using std::sin;
241  using std::cos;
242  Eigen::Matrix<PrimType_, 3, 3> mat = Eigen::Matrix<PrimType_, 3, 3>::Zero();
243  const PrimType_ x = rotation.x();
244  const PrimType_ y = rotation.y();
245  const PrimType_ z = rotation.z();
246  const PrimType_ dx = this->x();
247  const PrimType_ dy = this->y();
248  const PrimType_ dz = this->z();
249  const PrimType_ t2 = cos(y);
250  KINDR_ASSERT_TRUE(std::runtime_error, t2 != PrimType_(0), "Gimbal lock: cos(y) is zero!");
251  const PrimType_ t3 = 1.0/t2;
252  const PrimType_ t4 = cos(z);
253  const PrimType_ t5 = 1.0/(t2*t2);
254  const PrimType_ t6 = sin(y);
255  const PrimType_ t7 = sin(z);
256  const PrimType_ t8 = t6*t6;
257  mat(0,0) = -dz*t3*t7+dy*t4*t5*t6;
258  mat(0,1) = -dz*t3*t4-dy*t5*t6*t7;
259  mat(1,0) = dz*t4;
260  mat(1,1) = -dz*t7;
261  mat(2,0) = -dy*(t4+t4*t5*t8)+dz*t3*t6*t7;
262  mat(2,1) = dy*(t7+t5*t7*t8)+dz*t3*t4*t6;
263 
264  return mat;
265  }

267  Eigen::Matrix<PrimType_, 3, 3> getMappingFromSecondDiffToLocalAngularVelocity(const EulerAnglesXyz<PrimType_>& rotation) const {
268  using std::sin;
269  using std::cos;
270  Eigen::Matrix<PrimType_, 3, 3> mat = Eigen::Matrix<PrimType_, 3, 3>::Zero();
271  const PrimType_ x = rotation.x();
272  const PrimType_ y = rotation.y();
273  const PrimType_ z = rotation.z();
274  const PrimType_ dx = this->x();
275  const PrimType_ dy = this->y();
276  const PrimType_ dz = this->z();
277  const PrimType_ t2 = cos(z);
278  const PrimType_ t3 = cos(y);
279  const PrimType_ t4 = sin(y);
280  const PrimType_ t5 = sin(z);
281  mat(0,0) = -dy*t2*t4-dz*t3*t5;
282  mat(0,1) = dz*t2;
283  mat(1,0) = dy*t4*t5-dz*t2*t3;
284  mat(1,1) = -dz*t5;
285  mat(2,0) = dy*t3;

289  Eigen::Matrix<PrimType_, 3, 3> getMappingFromGlobalAngularVelocityToSecondDiff(const EulerAnglesXyz<PrimType_>& rotation) const {
290  using std::sin;
291  using std::cos;
292  Eigen::Matrix<PrimType_, 3, 3> mat = Eigen::Matrix<PrimType_, 3, 3>::Zero();
293  const PrimType_ x = rotation.x();
294  const PrimType_ y = rotation.y();
295  const PrimType_ z = rotation.z();
296  const PrimType_ dx = this->x();
297  const PrimType_ dy = this->y();
298  const PrimType_ dz = this->z();
299  const PrimType_ t2 = sin(x);
300  const PrimType_ t3 = sin(y);
301  const PrimType_ t4 = cos(y);
302  const PrimType_ t5 = cos(x);
303  KINDR_ASSERT_TRUE(std::runtime_error, t4 != PrimType_(0), "Gimbal lock: cos(y) is zero!");
304  const PrimType_ t6 = 1.0/(t4*t4);
305  const PrimType_ t7 = t3*t3;
306  const PrimType_ t8 = 1.0/t4;
307  mat(0,1) = dy*(t2+t2*t6*t7)+dx*t3*t5*t8;
308  mat(0,2) = -dy*(t5+t5*t6*t7)+dx*t2*t3*t8;
309  mat(1,1) = -dx*t2;
310  mat(1,2) = dx*t5;
311  mat(2,1) = -dx*t5*t8-dy*t2*t3*t6;
312  mat(2,2) = -dx*t2*t8+dy*t3*t5*t6;
313 
314  return mat;
315  }

317  Eigen::Matrix<PrimType_, 3, 3> getMappingFromSecondDiffToGlobalAngularVelocity(const EulerAnglesXyz<PrimType_>& rotation) const {
318  using std::sin;
319  using std::cos;
320  Eigen::Matrix<PrimType_, 3, 3> mat = Eigen::Matrix<PrimType_, 3, 3>::Zero();
321  const PrimType_ x = rotation.x();
322  const PrimType_ y = rotation.y();
323  const PrimType_ z = rotation.z();
324  const PrimType_ dx = this->x();
325  const PrimType_ dy = this->y();
326  const PrimType_ dz = this->z();
327  const PrimType_ t2 = cos(y);
328  const PrimType_ t3 = sin(x);
329  const PrimType_ t4 = cos(x);
330  const PrimType_ t5 = sin(y);
331  mat(0, 2) = dy*t2;
332  mat(1, 1) = -dx*t3;
333  mat(1, 2) = -dx*t2*t4+dy*t3*t5;
334  mat(2, 1) = dx*t4;
335  mat(2, 2) = -dx*t2*t3-dy*t4*t5;
336  return mat;
337  }
