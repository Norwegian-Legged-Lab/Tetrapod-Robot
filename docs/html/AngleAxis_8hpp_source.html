259  AngleAxis getUnique() const {
260  const Scalar anglePosNegPi = kindr::wrapPosNegPI(angle());
261  AngleAxis aa(anglePosNegPi, axis()); // first wraps angle into [-pi,pi)
262  if(aa.angle() > 0.0) {
263  return aa;
264  }
265  else if(aa.angle() < 0.0) {
266  if(aa.angle() != -M_PI) {
267  return AngleAxis(-aa.angle(),-aa.axis());
268  }
269  else { // angle == -pi, so axis must be viewed further, because -pi,axis does the same as -pi,-axis
270  if(aa.axis()[0] < 0.0) {
271  return AngleAxis(-aa.angle(),-aa.axis());
272  }
273  else if(aa.axis()[0] > 0) {
274  return AngleAxis(-aa.angle(),aa.axis());
275  }
276  else { // v1 == 0
277 
278  if(aa.axis()[1] < 0.0) {
279  return AngleAxis(-aa.angle(),-aa.axis());
280  }
281  else if(aa.axis()[1] > 0) {
282  return AngleAxis(-aa.angle(),aa.axis());
283  }
284  else { // v2 == 0
285 
286  if(aa.axis()[2] < 0.0) { // v3 must be -1 or 1
287  return AngleAxis(-aa.angle(),-aa.axis());
288  }
289  else {
290  return AngleAxis(-aa.angle(),aa.axis());
291  }
292  }
293  }
294  }
295  }
296  else { // angle == 0
297  return AngleAxis();
298  }
299  }

364 template<typename DestPrimType_, typename SourcePrimType_>
365 class ConversionTraits<AngleAxis<DestPrimType_>, RotationVector<SourcePrimType_>> {
366  public:
367  inline static AngleAxis<DestPrimType_> convert(const RotationVector<SourcePrimType_>& rotationVector) {
368  typedef typename RotationVector<SourcePrimType_>::Scalar Scalar;
369 
370  // Cast to destination primitive type
371  const Eigen::Matrix<DestPrimType_, 3, 1> vector = rotationVector.toImplementation().template cast<DestPrimType_>();
372  const DestPrimType_ norm = vector.norm();
373 
374  // Check for identity rotation (we cannot divide by zero norm).
375  if (norm < internal::NumTraits<Scalar>::dummy_precision()) {
376  return AngleAxis<DestPrimType_>();
377  }
378  return AngleAxis<DestPrimType_>(norm, vector.normalized());
379  }
380 };
