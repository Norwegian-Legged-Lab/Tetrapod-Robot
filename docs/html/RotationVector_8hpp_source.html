323 template<typename DestPrimType_, typename SourcePrimType_>
324 class ConversionTraits<RotationVector<DestPrimType_>, RotationQuaternion<SourcePrimType_>> {
325  public:
326  inline static RotationVector<DestPrimType_> convert(const RotationQuaternion<SourcePrimType_>& q) {
327  using std::acos;
328  using std::sqrt;
329  using std::atan2;
330 
331  const DestPrimType_ imaginaryVectorNormSquared = DestPrimType_(1.0)-q.real()*q.real();
332 
333  if (imaginaryVectorNormSquared < internal::NumTraits<SourcePrimType_>::dummy_precision()) {
334  if (q.real() > 0.0) {
335  return RotationVector<DestPrimType_>(DestPrimType_(2.0)*q.imaginary().template cast<DestPrimType_>());
336  }
337  else {
338  return RotationVector<DestPrimType_>(DestPrimType_(-2.0)*q.imaginary().template cast<DestPrimType_>());
339  }
340  }
341  const DestPrimType_ imaginaryVectorNorm = q.imaginary().norm();
342  return RotationVector<DestPrimType_>((DestPrimType_(2.0)*atan2(imaginaryVectorNorm, q.real())/imaginaryVectorNorm*q.imaginary()).template cast<DestPrimType_>());
343  }
344 };
