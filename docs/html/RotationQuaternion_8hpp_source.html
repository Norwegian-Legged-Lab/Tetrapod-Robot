340  RotationQuaternion getUnique() const {
341  if(this->w() > 0) {
342  return *this;
343  } else if (this->w() < 0){
344  return RotationQuaternion(-this->w(),-this->x(),-this->y(),-this->z());
345  } else { // w == 0
346 
347  if(this->x() > 0) {
348  return *this;
349  } else if (this->x() < 0){
350  return RotationQuaternion(-this->w(),-this->x(),-this->y(),-this->z());
351  } else { // x == 0
352 
353  if(this->y() > 0) {
354  return *this;
355  } else if (this->y() < 0){
356  return RotationQuaternion(-this->w(),-this->x(),-this->y(),-this->z());
357  } else { // y == 0
358 
359  if(this->z() > 0) { // z must be either -1 or 1 in this case
360  return *this;
361  } else {
362  return RotationQuaternion(-this->w(),-this->x(),-this->y(),-this->z());
363  }
364  }
365  }

373  Eigen::Matrix<PrimType_,4,4> getQuaternionMatrix() const {
374  Eigen::Matrix<PrimType_,4,4> Qleft;
375  Qleft(0,0) = w(); Qleft(0,1) = -x(); Qleft(0,2) = -y(); Qleft(0,3) = -z();
376  Qleft(1,0) = x(); Qleft(1,1) = w(); Qleft(1,2) = -z(); Qleft(1,3) = y();
377  Qleft(2,0) = y(); Qleft(2,1) = z(); Qleft(2,2) = w(); Qleft(2,3) = -x();
378  Qleft(3,0) = z(); Qleft(3,1) = -y(); Qleft(3,2) = x(); Qleft(3,3) = w();
379 
380  return Qleft;
381  }

387  Eigen::Matrix<PrimType_,4,4> getConjugateQuaternionMatrix() const {
388  Eigen::Matrix<PrimType_,4,4> Qright;
389  Qright(0,0) = w(); Qright(0,1) = -x(); Qright(0,2) = -y(); Qright(0,3) = -z();
390  Qright(1,0) = x(); Qright(1,1) = w(); Qright(1,2) = z(); Qright(1,3) = -y();
391  Qright(2,0) = y(); Qright(2,1) = -z(); Qright(2,2) = w(); Qright(2,3) = x();
392  Qright(3,0) = z(); Qright(3,1) = y(); Qright(3,2) = -x(); Qright(3,3) = w();
393 
394  return Qright;

400  Eigen::Matrix<PrimType_,3,4> getGlobalQuaternionDiffMatrix() const {
401  Eigen::Matrix<PrimType_,3,4> H;
402  H(0,0) = -x(); H(0,1) = w(); H(0,2) = -z(); H(0,3) = y();
403  H(1,0) = -y(); H(1,1) = z(); H(1,2) = w(); H(1,3) = -x();

405  return H;
