353  EulerAnglesZyx getUnique() const {
354  Base zyx(kindr::floatingPointModulo(z()+M_PI,2*M_PI)-M_PI,
355  kindr::floatingPointModulo(y()+M_PI,2*M_PI)-M_PI,
356  kindr::floatingPointModulo(x()+M_PI,2*M_PI)-M_PI); // wrap all angles into [-pi,pi)
357 
358  const double tol = 1e-3;
359 
360  // wrap angles into [-pi,pi),[-pi/2,pi/2),[-pi,pi)
361  if(zyx.y() < -M_PI/2 - tol)
362  {
363  if(zyx.x() < 0) {
364  zyx.x() = zyx.x() + M_PI;
365  } else {
366  zyx.x() = zyx.x() - M_PI;
367  }
368 
369  zyx.y() = -(zyx.y() + M_PI);
370 
371  if(zyx.z() < 0) {
372  zyx.z() = zyx.z() + M_PI;
373  } else {
374  zyx.z() = zyx.z() - M_PI;
375  }
376  }
377  else if(-M_PI/2 - tol <= zyx.y() && zyx.y() <= -M_PI/2 + tol)
378  {
379  zyx.x() -= zyx.z();
380  zyx.z() = 0;
381  }
382  else if(-M_PI/2 + tol < zyx.y() && zyx.y() < M_PI/2 - tol)
383  {
384  // ok
385  }
386  else if(M_PI/2 - tol <= zyx.y() && zyx.y() <= M_PI/2 + tol)
387  {
388  // todo: M_PI/2 should not be in range, other formula?
389  zyx.x() += zyx.z();
390  zyx.z() = 0;
391  }
392  else // M_PI/2 + tol < zyx.y()
393  {
394  if(zyx.x() < 0) {
395  zyx.x() = zyx.x() + M_PI;
396  } else {
397  zyx.x() = zyx.x() - M_PI;
398  }
399 
400  zyx.y() = -(zyx.y() - M_PI);
401 
402  if(zyx.z() < 0) {
403  zyx.z() = zyx.z() + M_PI;
404  } else {
405  zyx.z() = zyx.z() - M_PI;
406  }
407  }
408 
409  return EulerAnglesZyx(zyx);
410  }

421  typename Eigen::Matrix<PrimType_, 3, 3> getMappingFromDiffToLocalAngularVelocity() const {
422  using std::sin;
423  using std::cos;
424  Eigen::Matrix<PrimType_, 3, 3> mat = Eigen::Matrix<PrimType_, 3, 3>::Zero();
425  const PrimType_ x = this->x();
426  const PrimType_ y = this->y();
427  const PrimType_ z = this->z();
428  const PrimType_ t2 = cos(x);
429  const PrimType_ t3 = cos(y);
430  const PrimType_ t4 = sin(x);
431  mat(0, 0) = -sin(y);
432  mat(0, 2) = 1.0;
433  mat(1, 0) = t3*t4;
434  mat(1, 1) = t2;
435  mat(2, 0) = t2*t3;
436  mat(2, 1) = -t4;
437  return mat;
438  }

440  typename Eigen::Matrix<PrimType_, 3, 3> getMappingFromLocalAngularVelocityToDiff() const {
441  using std::sin;
442  using std::cos;
443  Eigen::Matrix<PrimType_, 3, 3> mat = Eigen::Matrix<PrimType_, 3, 3>::Zero();
444  const PrimType_ x = this->x();
445  const PrimType_ y = this->y();
446  const PrimType_ z = this->z();
447  const PrimType_ t2 = cos(y);
448  const PrimType_ t3 = 1.0/t2;
449  KINDR_ASSERT_TRUE(std::runtime_error, t2 != PrimType_(0.0), "Gimbal lock: cos(y) is zero!");
450  const PrimType_ t4 = cos(x);
451  const PrimType_ t5 = sin(x);
452  const PrimType_ t6 = sin(y);
453  mat(0,1) = t3*t5;
454  mat(0,2) = t3*t4;
455  mat(1,1) = t4;
456  mat(1,2) = -t5;
457  mat(2,0) = 1.0;
458  mat(2,1) = t3*t5*t6;
459  mat(2,2) = t3*t4*t6;
460  return mat;
461  }
