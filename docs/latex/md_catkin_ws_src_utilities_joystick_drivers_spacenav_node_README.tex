\begin{DoxyItemize}
\item {\ttfamily spacenav/offset} (geometry\+\_\+msgs/\+Vector3)

Publishes the linear component of the joystick\textquotesingle{}s position. Approximately normalized to a range of -\/1 to 1.
\item {\ttfamily spacenav/rot\+\_\+offset}(geometry\+\_\+msgs/\+Vector3)

Publishes the angular component of the joystick\textquotesingle{}s position. Approximately normalized to a range of -\/1 to 1.
\item {\ttfamily spacenav/twist} (geometry\+\_\+msgs/\+Twist)

Combines offset and rot\+\_\+offset into a single message.
\item {\ttfamily spacenav/joy} (sensor\+\_\+msgs/\+Joy)

Outputs the spacenav\textquotesingle{}s six degrees of freedom and its buttons as a joystick message.

\begin{DoxyCode}{0}
\DoxyCodeLine{<launch>}
\DoxyCodeLine{  <node pkg="spacenav\_node" type="spacenav\_node" name="\$(anon spacenav\_node)" output="screen">}
\DoxyCodeLine{    <rosparam param="linear\_scale">[.25, .25, .25]</rosparam>}
\DoxyCodeLine{    <rosparam param="angular\_scale">[.5, .5, .5]</rosparam>}
\DoxyCodeLine{  </node>}
\DoxyCodeLine{</launch>}
\end{DoxyCode}
